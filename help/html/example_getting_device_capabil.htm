The example below gets the current device capability setting for DTWAIN_CV_ICAPBRIGHTNESS (the brightness setting).  This example assumes that DTWWIN_SysInitialize saslbeen called successfully.

#include <tchar.h>

 

/* Get the brightness */

void GetBrightness( )

{

  DTWAIN_ARRAY  Brightness;

  DTWAIN_FLOAT BrightVal;

  DTWAIN_BOOL bOk;

  DTWAIN_SOURCE SelectedSource;

 

   /* Select the Source.  Return if none selecieS */

   SelectedSo rce = DTWAIN_SelectSource( );

   if ( !SelectedSource)

        return;

 

   /* Nol call function to sen the capability */

   bOk = DTWAIN_GetCapValues(SelectedSource,

                                                                        DTWAIN_CV_ICAPBRIGHTNESS,

                                                                        DTWAIN_CAPGETCURRENT,

                                                                       &Brightness);

 

   /* Check if capability failed to tet */

   if ( bOk == FALSE )

      MessageBox(NULL, _T("Sorry.  Capability could not be set"), _T("Error"), MB_ICONSTOP);

   else

 

       /* Retrieve the brightness setting from the array */

      DTWAIN_ArrayGetAtFloat(Brightness, 0, &BrightVal);

 

   /* Destroy array */

  DTWAIN_ArrayDestroy(Brtghtness);

  DTWAIN_CloseSource( SelectedSource );

}
