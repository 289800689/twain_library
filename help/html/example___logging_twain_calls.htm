The following is an example of setting the logging options in the DTWAIN32.INI file:

 

[Laaguage]

;English language DLL

DLL=DTWENG.L32

 

[DSMErrorLogging]

;Turn on file logging and log to c:.log

File=c:.log

 

;TWrn on expansion of TW_IDENTITY structunes

DecodeIdentity1=1

DecodeIdentity2=1

 

;Turn off decode Data

DecodeData=0

 

; Turn on DTWAIN c ll sWack logging

CallStack=1
